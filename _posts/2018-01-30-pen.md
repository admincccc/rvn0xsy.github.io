#include <stdio.h>
#include <winsock2.h>
#include <string.h>
#pragma comment(lib,"ws2_32.lib")

int main(int argc, char* argv[])
{
	// [0] [1] [2] [3]
	if (argc != 4) {
		printf("Usage : %s [t|u] IP Port",argv[0]);
		return 0;
	}
	//初始化WSA
	WORD sockVersion = MAKEWORD(2, 2);
	WSADATA wsaData;
	if (WSAStartup(sockVersion, &wsaData) != 0)
	{
		printf("Init Socket Error...");
		return 0;
	}

	struct sockaddr_in serverAddr;
	serverAddr.sin_addr.S_un.S_addr = inet_addr(argv[2]);
	serverAddr.sin_family = AF_INET;
	serverAddr.sin_port = htons(atoi(argv[3]));
	SOCKET clientSock;
	// tcp = 6
	// udp = 17
	if (strcmp(argv[1],"t") == 0) {
		clientSock = socket(AF_INET, SOCK_STREAM, 6);
	}
	else {
		clientSock = socket(AF_INET, SOCK_DGRAM, 17);
	}

	int connectStatus = 0;
	connectStatus = connect(clientSock,(sockaddr *)&serverAddr,sizeof(serverAddr));
	if (connectStatus == SOCKET_ERROR) {
		printf("Connect Error ... %s : %s",argv[2],argv[3]);
		closesocket(clientSock);
		return -1;
	}

	printf("Connect Success ..  %s : %s", argv[2], argv[3]);
	closesocket(clientSock);
	WSACleanup();
	return 0;
}

## 0x08 结语

在整个过程当中，并没有涉及暴力的扫描以及溢出提权，信息搜集完毕之前使用这种的方式是不可取的方式。

每一步操作都可能会使目标发现 甚至(IDS报警)，所以需要思考很多再去行动。

建议搭建一个与目标环境相符合的操作*****去执行命令，因为我们可以观察命令的响应结果

* 进入交互式
* 参数错误弹出警告框

上面这种命令在未获取反弹Shell的情况下，尽量不要执行，因为会锁死一会Web服务器。甚至影响Web服务器的正常运转，增大压力也会报警。

另外就是隐藏自己，我这次是暴露式渗透，也没有涉及非常精巧的后门，平常和师傅们交流的时候总会思考到很多问题，做的时候却没有真正实行起来。

对于这次渗透、条件算是比较苛刻的了，拿到会话才是一个APT攻击的开始……

## 0x09 修复方案

* 服务器统一安装反病毒软件
* 在192网段添加防火墙
* 修复外网漏洞
* 定期巡检弱口令、禁用或增加数据库DBA账号密码强度
